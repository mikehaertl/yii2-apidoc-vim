
*yii/console/widgets/Table*   >> |yii/base/Widget|
===========================

DESCRIPTION
------------------------------------------------------------------------------

 For example,

   $table = new Table();

   echo $table
       ->setHeaders(['test1', 'test2', 'test3'])
       ->setRows([
           ['col1', 'col2', 'col3'],
           ['col1', 'col2', ['col3-0', 'col3-1', 'col3-2']],
       ])
       ->run();

 or

 '''php echo Table::widget([     'headers' => ['test1', 'test2', 'test3'],    
 'rows' => [         ['col1', 'col2', 'col3'],         ['col1', 'col2',
 ['col3-0', 'col3-1', 'col3-2'',     ], ]);
